export class TextureLoader {
  public static use(
    resource: ILoaderResource,
    next: (...args: any[]) => void
  ): void {
    if (resource.data && resource.type === Resource.TYPE.IMAGE) {
      resource.texture = Texture.fromLoader(
        resource.data,
        resource.url,
        resource.name
      );
    }
    next();
  }
}
